   171→function formatDate(dateStr: string) {
   172→  try {
   173→    return new Date(dateStr).toLocaleDateString("en-IN", {
   174→      day: "2-digit",
   175→      month: "short",
   176→      year: "numeric",
   177→    });
   178→  } catch {
   179→    return dateStr;
   180→  }
   181→}
   182→
   183→function formatDateTime(dateStr: string) {
   184→  try {
   185→    return new Date(dateStr).toLocaleString("en-IN", {
   186→      day: "2-digit",
   187→      month: "short",
   188→      year: "numeric",
   189→      hour: "2-digit",
   190→      minute: "2-digit",
   191→    });
   192→  } catch {
   193→    return dateStr;
   194→  }
   195→}
   196→
   197→function formatPrice(price: string) {
   198→  const num = parseFloat(price.replace(/[^0-9.]/g, ""));
   199→  if (isNaN(num)) return price;
   200→  return "₹" + num.toLocaleString("en-IN");
   201→}
